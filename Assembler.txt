_MAX dt 604462909807314587353087
@aux1 dt ?
@aux2 dt ?
@aux3 dt ?
a@Main dt ?

i@Main dt ?
_1.5000E+4 dt 1.5000E+4
.code
overflow:
invoke MessageBox, NULL, addr errorOverflowSuma, addr mensaje, MB_OK

divcero:
invoke MessageBox, NULL, addr errorDivisionCero, addr mensaje, MB_OK

main:
FLD _3.45000e+5
FADD _1.5000e+4
FSTP @aux1
