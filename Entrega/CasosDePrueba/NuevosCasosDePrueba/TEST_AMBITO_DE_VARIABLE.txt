INTEGER i;
FLOAT f;

%% EL PROC D TOMA LAS VAR ii Y ff QUE SON LOCALES AL PROC A Y TOMA i DEL MAIN (ii E i EN ESTE CASO SON LA MISMA PERO TIENE DIFERENTE AMBITO, ES EN ESTE CASO PORQUE DENTRO DEL PROC A PUEDE QUE CAMBIE DE VALOR)
PROC a(INTEGER ii, FLOAT ff) NA=3_i, NS=3_i{
	PROC b() NA=2_i, NS=2_i{
		PROC c() NA=1_i, NS=2_i{
			PROC d() NA=0_i, NS=2_i{
				ff = ii + i;
			}
		}
	}
}

a(i,f);
